    notes1 = map(Note, notes)

from music21 import midi
def playMidi(filename):
    mf = midi.MidiFile()
    mf.open(filename)
    mf.read()
    mf.close()
    s = midi.translate.midiFileToStream(mf)
    s.show('midi')
    
playMidi("tune1.midi")
